\section{Basic 2D Function}
\begin{tikzpicture}
  \begin{axis}[axis lines=middle, xmin=-2, xmax=2, ymin=-2, ymax=2,
               xlabel={$x$}, ylabel={$y$}, title={Cubic}]
    \addplot[color=red, thick, samples=100, domain=-2:2] {x^3};
  \end{axis}
\end{tikzpicture}

\section{Intermediate 2D Plot: Multiple Functions}
% Demonstrates: multiple plots, styling, domain, legend, and text annotation
\begin{tikzpicture}
  \begin{axis}[
    axis lines=middle,
    xmin=-3, xmax=3,
    ymin=-1, ymax=5,
    xlabel={$x$}, ylabel={$y$},
    title={Multiple Functions with Styling},
    legend pos=north west,
    grid=major
  ]
    % Quadratic function with solid red line
    \addplot[color=red, thick, samples=100, domain=-3:3] {x^2};
    \addlegendentry{$x^2$}
    
    % Exponential function with dashed blue line
    \addplot[color=blue, dashed, thick, samples=100, domain=-3:1.5] {exp(x/2)};
    \addlegendentry{$e^{x/2}$}
    
    % Sine function with markers, limited domain
    \addplot[color=green!60!black, mark=*, mark size=1.5pt, samples=50, domain=0:3] {2*sin(deg(x)) + 2};
    \addlegendentry{$2\sin(x) + 2$}
    
    % Text annotation at a specific point
    \node[pin=45:{Local maximum}] at (axis cs: 0, 0) {};
  \end{axis}
\end{tikzpicture}

\section{Custom Ticks and Grid}
\begin{tikzpicture}
  \begin{axis}[axis lines=middle, xmin=0, xmax=3.2, ymin=-1.5, ymax=1.5,
               xtick={0, 1.57, 3.14},
               xticklabels={0, $\frac{\pi}{2}$, $\pi$},
               ytick={-1,0,1}, xmajorgrids=true,
               title={Sine with Custom Ticks}]
    \addplot[blue, thick, samples=150, domain=0:3.14] {sin(deg(x))};
  \end{axis}
\end{tikzpicture}

\section{Stacked Bar Chart}
\begin{tikzpicture}
  \begin{axis}[ybar stacked, ymin=0, ymax=10,
               xlabel={Category}, ylabel={Count},
               symbolic x coords={A,B,C}, xtick=data,
               title={Stacked Bars}]
    \addplot coordinates {(A,2) (B,3) (C,4)};
    \addplot coordinates {(A,4) (B,1) (C,2)};
  \end{axis}
\end{tikzpicture}

\section{3D Surface}
\begin{tikzpicture}
  \begin{axis}[view={60}{30}, title={Saddle}]
    \addplot3[surf, samples=30, domain=-2:2, domain y=-2:2] {x^2 - y^2};
  \end{axis}
\end{tikzpicture}
