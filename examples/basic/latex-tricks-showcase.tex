    \PD{f}{x} &= \lim_{h\to 0} \frac{f(x+h)-f(x)}{h},\\

\section{Conditional Content}
Toggle solutions on or off with a single switch in the preamble.
\begin{enumerate}
    \item Compute $\int_0^1 x^2\,dx$.
    \ifsolution
    
    Solution: Using $\int x^2\,dx = x^3/3$, we get $1/3$.
    \fi
    \item Solve $f'(x)=2x$.
    \ifsolution
    
    Solution: $f(x)=x^2 + C$.
    \fi
\end{enumerate}
