class GNN_Transformer(tgnn.MessagePassing):
    def __init__(self, in_channels, out_channels):
        super(GNN_Transformer, self).__init__(aggr='add')
        self.lin = torch.nn.Linear(in_channels, out_channels)

        # 1. Message Passing
