%----------------------------------------------------------------------------------------
%	PACKAGES AND OTHER DOCUMENT CONFIGURATIONS
%----------------------------------------------------------------------------------------

\usepackage[a4paper, hmargin=25mm, vmargin=30mm, top=20mm]{geometry} % Use A4 paper and set margins

\usepackage{fancyhdr} % Customize the header and footer

\usepackage{lastpage} % Required for calculating the number of pages in the document

\usepackage{hyperref} % Colors for links, text and headings

\setcounter{secnumdepth}{0} % Suppress section numbering

\usepackage[utf8]{inputenc} % Required for inputting international characters
\usepackage[T1]{fontenc} % Output font encoding for international characters

\usepackage{color} % Required for custom colors
\definecolor{slateblue}{rgb}{0.17,0.22,0.34}

\usepackage{sectsty} % Allows customization of titles
\sectionfont{\color{slateblue}} % Color section titles

\pagestyle{fancy}
\fancyhf{} % Clear default headers/footers
\renewcommand{\headrulewidth}{0pt} % No header rule
\renewcommand{\footrulewidth}{0pt} % No footer rule
\fancyfoot[C]{\thepage\ of \pageref{LastPage}} % Page numbers in footer

%----------------------------------------------------------------------------------------
%	SECTION TITLE STYLING AND SPACING
%----------------------------------------------------------------------------------------

\usepackage{titlesec} % Required for modifying sections

%------------------------------------------------
% Section
\titleformat{\section} % Section type being modified
{\color{slateblue}\Large\bfseries\scshape\raggedright} % Format of the whole section
{}{0em} % Format of the section label
{} % Code before the label
[\color{black}\titlerule] % Code after the label

\titlespacing{\section}{0pt}{12pt}{5pt} % Spacing around section titles, the order is: left, before and after

%------------------------------------------------
% Subsection
\titleformat{\subsection} % Section type being modified
{\large\bfseries\scshape\raggedright} % Format of the whole section
{}{0em} % Format of the section label
{} % Code before the label
[\color{black}] % Code after the label

\titlespacing{\subsection}{0pt}{8pt}{3pt} % Spacing around section titles, the order is: left, before and after

	\hspace{2cm} \= \hspace{10cm} \= \kill

	\>\>\small{\href{#4}{#4}}\\[4pt]
	\>\textbf{\textit{#5}}\\
	\>\small{#6}
	\end{tabbing}
	\vspace{2mm}
}
